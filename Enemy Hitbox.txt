local highlightColor = Color3.new(1, 0, 0)

local function processObject(object)
    if object:IsA("MeshPart") and object.Name == "Head" then
        object.Size = Vector3.new(20, 20, 20)

        local highlight = Instance.new("Highlight")
        highlight.Parent = object
        highlight.FillColor = highlightColor
        highlight.OutlineColor = highlightColor
    end
end

for _, child in pairs(game.Workspace:GetDescendants()) do
    processObject(child)
end

game.Workspace.DescendantAdded:Connect(function(child)
    processObject(child)
end)
